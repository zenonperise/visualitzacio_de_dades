import seaborn as sns
from matplotlib import pyplot as plt

fig, ax = plt.subplots(figsize=(10,5))
ax.set_title("Valor bursatil de Twitter. Mxim diari.")

plot = sns.swarmplot(y='High', data=df, ax=ax, )
plot.set_ylabel('Valor ($ per Acci)')
plt.box(False)
